var calc = require('./calc.js');

import calc from './calc.js'
//只获取到calc.js中的add方法(按需获取)

    # 用export 和import 的写法注意点

    1、如果模块中使用 export default {}
    只能通过 import 对象名称 from '模块路径'
    不能通过 import {对象名称} from '模块路径'
2、如果使用 import {对象名称} from '模块路径' 导出具体某个对象或者方法名称

    需要单独导出对象或者方法：
    export function add(){}
    export function sub(){}

    使用：
    import {add, sub} from '模块路径'
    直接使用 add,sub方法

    注意：不能直接使用    import 对象名称 from '模块路径'

    # 常量

    1. 在ES3中，没有常量这一说。
    2. 在ES5中，通过变量的定义只读，不可修改。
    3. 在ES6中，用==const==定义常量也是只读，不可修改。
    # 举个例子,设置圆周率PI的值：
    //es5写法
Object.defineProperty(window,'PI',{
    value:3.1415926,//定义属性值
    writable:false,//是否只读
})

//es6写法
const PI = 3.1415926;
console.log(PI)

    # 题外话：这里关于let和var、const区别：

    1. 使用var声明的变量，其作用域为该语句所在的函数内，且存在变量提升现象；
    2. 使用let声明的变量，其作用域为该语句所在的代码块内，不存在变量提升；
    3. 使用const声明的是常量，在后面出现的代码中不能再修改该常量的值。

    # 属性名表达式

    # JavaScript 定义对象的属性，有两种方法。
    // 方法一
obj.foo = true;

// 方法二
obj['a' + 'bc'] = 123;
